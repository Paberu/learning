// три примера из DynamicArray. Числовые значения встречаются часто, лучше заменить их на константы с говорящими переменными

// было:
    def hash1(self, str1):
        # 17
        code = 0
        for c in str1:
            code = (code * 17 + ord(c)) % self.filter_len
        return int(code)

    def hash2(self, str1):
        # 223
        code = 0
        for c in str1:
            code = (code * 223 + ord(c)) % self.filter_len
        return int(code)
		
//стало:
    def get_hash(self, value, hash_code):
        code = 0
        for c in value:
            code = (code * hash_code + ord(c)) % self.filter_len
        return int(code)
// пример из BloomFilter. Можно отказаться от заданных числовых значений и помещать их в self.hash_code1, self.hash_code2
или вовсе отказаться от заданного количества констант и хранить их в списке self.hash_codes, размер и содержимое которого будут 
рассчитываться исходя из данных, которые будут проведены сквозь фильтр. Тогда все числовые значения будут храниться в одном месте.
